# Ranking Tab Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a Ranking tab where users can manually order players via drag-and-drop; this order propagates to the Stats and H2H tabs.

**Architecture:** `rank_position INTEGER` added to the `players` table. New players auto-append. A new `PUT /projects/:id/ranking` endpoint replaces the full order atomically. `GET /players` and `GET /stats` both return data ordered by `rank_position ASC`. The Ranking page keeps local state until the user clicks Save.

**Tech Stack:** Rust/Axum/sqlx (backend), SvelteKit + Svelte 5 runes + `svelte-dnd-action` (frontend)

// ── Reorder players ───────────────────────────────────────────────────────────

#[derive(Deserialize)]
struct ReorderRequest {
    player_ids: Vec<Uuid>,
}

async fn reorder_players(
    State(state): State<AppState>,
    AuthUser(user): AuthUser,
    Path(project_id): Path<Uuid>,
    Json(body): Json<ReorderRequest>,
) -> Result<impl IntoResponse> {
    require_project(&state.db, project_id, user.id).await?;

    let existing_ids: Vec<Uuid> = sqlx::query_scalar!(
        "SELECT id FROM players WHERE project_id = $1",
        project_id,
    )
    .fetch_all(&state.db)
    .await?;

    let existing_set: std::collections::HashSet<Uuid> =
        existing_ids.into_iter().collect();

    if body.player_ids.len() != existing_set.len() {
        return Err(AppError::UnprocessableEntity(
            "player_ids must contain exactly all players in this project".into(),
        ));
    }

    for &pid in &body.player_ids {
        if !existing_set.contains(&pid) {
            return Err(AppError::UnprocessableEntity(
                "player_ids contains an id not in this project".into(),
            ));
        }
    }

    let mut tx = state.db.begin().await?;
    for (i, &player_id) in body.player_ids.iter().enumerate() {
        sqlx::query!(
            "UPDATE players SET rank_position = $1 WHERE id = $2 AND project_id = $3",
            (i + 1) as i32,
            player_id,
            project_id,
        )
        .execute(&mut *tx)
        .await?;
    }
    tx.commit().await?;

    Ok(StatusCode::OK)
}
```

        b_rate
            .partial_cmp(&a_rate)

To:

    // Preserve rank_position order from the players query above.

export function makeApi(fetchFn: typeof fetch, baseUrl: string, sessionId?: string | null) {
	async function req(method: string, path: string, body?: unknown): Promise<Response> {
		const headers: Record<string, string> = {};
		if (sessionId) headers['Cookie'] = `session_id=${sessionId}`;
		if (body !== undefined) headers['Content-Type'] = 'application/json';
		return fetchFn(baseUrl + path, {
			method,
			credentials: 'include',
			headers,
			body: body !== undefined ? JSON.stringify(body) : undefined
		});
	}

	return {
		get: (path: string) => req('GET', path),
		post: (path: string, body?: unknown) => req('POST', path, body),
		patch: (path: string, body: unknown) => req('PATCH', path, body),
		put: (path: string, body: unknown) => req('PUT', path, body),
		delete: (path: string) => req('DELETE', path)
	};
}

	let saveSuccess = $state(false);

	const isDirty = $derived(items.map((i) => i.id).join(',') !== savedOrder.join(','));

	function handleConsider(e: CustomEvent<DndEvent<RankedPlayer>>) {
		items = e.detail.items;
	}

	function handleFinalize(e: CustomEvent<DndEvent<RankedPlayer>>) {
		items = e.detail.items;
	}

	function handleRankInput(index: number, raw: string) {
		const pos = parseInt(raw, 10);
		if (isNaN(pos) || pos < 1 || pos > items.length) return;
		const target = Math.min(pos, items.length) - 1;
		if (target === index) return;
		const next = [...items];
		const [moved] = next.splice(index, 1);
		next.splice(target, 0, moved);
		items = next;
	}

	async function save() {
		saving = true;
		try {
			const res = await fetch(`${PUBLIC_API_URL}/projects/${data.project.id}/ranking`, {
				method: 'PUT',
				credentials: 'include',
				headers: { 'Content-Type': 'application/json' },
				body: JSON.stringify({ player_ids: items.map((i) => i.id) })
			});
			if (res.ok) {
				savedOrder = items.map((i) => i.id);
				saveSuccess = true;
				setTimeout(() => {
					saveSuccess = false;
				}, 1500);
			}
		} finally {
			saving = false;
		}
	}
</script>

<div class="space-y-4">
	<div class="flex items-center justify-between">
		<h2 class="text-lg font-semibold">Ranking</h2>
		<div class="flex items-center gap-3">
			{#if isDirty && !saveSuccess}
				<span class="text-sm text-muted-foreground">Unsaved changes</span>
			{/if}
			<Button
				disabled={!isDirty || saving}
				onclick={save}
			>
				{#if saveSuccess}
					Saved ✓
				{:else if saving}
					Saving…
				{:else}
					Save
				{/if}
			</Button>
		</div>
	</div>

	{#if items.length === 0}
		<Empty.Root>
			<Empty.Header>
				<Empty.Title>No players yet</Empty.Title>
				<Empty.Description>Add players on the Players tab before ranking them.</Empty.Description>
			</Empty.Header>
		</Empty.Root>
	{:else}
		<div
			class="flex max-w-xl flex-col gap-1"
			use:dndzone={{ items, flipDurationMs: FLIP_MS, type: 'ranking' }}
			onconsider={handleConsider}
			onfinalize={handleFinalize}
		>
			{#each items as item, i (item.id)}
				<div
					animate:flip={{ duration: FLIP_MS }}
					class="flex items-center gap-3 rounded-md border border-border bg-background px-3 py-2"
				>
					<span class="cursor-grab text-muted-foreground select-none">⠿</span>
					<input
						type="text"
						value={i + 1}
						class="w-8 rounded border border-border bg-muted/40 px-1 py-0.5 text-center text-xs"
						onchange={(e) => handleRankInput(i, (e.currentTarget as HTMLInputElement).value)}
						onblur={(e) => handleRankInput(i, (e.currentTarget as HTMLInputElement).value)}
					/>
					<span class="flex-1 font-medium">{item.name}</span>
					{#if item.wins > 0 || item.losses > 0}
						<span class="text-xs text-muted-foreground">{item.wins}W · {item.losses}L</span>
						<span class="min-w-[2.5rem] text-right text-xs font-semibold">{item.winRate}</span>
					{/if}
				</div>
			{/each}
		</div>
		<p class="text-xs text-muted-foreground">Click the rank number to edit · Drag ⠿ to reorder</p>
	{/if}
</div>
```
